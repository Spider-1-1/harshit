    int insertion(int ar[],int size)
    {
    int i,j,temp,step=-1;
    for(i=1;i<size;i++)
    {
    temp=ar[i];
    j=i-1;
    while(temp<ar[j] && j>=0)
    {
    ar[j+1]=ar[j];
    j--;
    step++;
    }
    ar[j+1]=temp;
    }
    return step;
    }
    int main()
    {
    clock_t start=clock();
    int ar[10],i,step;
    clock_t end;
    double cpu_time_used;

    for(i=0;i<10;i++)
    {
    printf("Enter the array at location %d\t",i+1);
    scanf("%d",&ar[i]);

    step=insertion(ar,10);
    printf("Sorted Arrays are\n");
    for(i=0;i<10;i++)
    {
    printf("%d\n",ar[i]);
    }
    if(step==-1)
    {
    printf("Best complexity for selection sort\n");
    }
    else
    {
    printf("Worst complexity for selection sort\n");
    }
    end=clock();
    cpu_time_used=((double)(end-start))/CLOCKS_PER_SEC;
    printf("Time taken for search: %lf seconds\n",cpu_time_used);
    }
