int selection(int ar[], int size, int index) {
	int j,temp;
    if (index == size - 1) {
        return 0;
    }
    for ( j = index + 1; j < size; j++) {
        if (ar[index] > ar[j]) {
            temp = ar[index];
            ar[index] = ar[j];
            ar[j] = temp;
        }
    }
    return selection(ar, size, index + 1);
}
int main() {

	end=clock();
	cpu_time_used=((double)(end-start))/CLOCKS_PER_SEC;
	printf("Time taken for search: %lf seconds\n",cpu_time_used);
}
