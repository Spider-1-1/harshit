void mergeSort(char str[][10], int l, int r) {
    if (l < r) {
        int m = l + (r - l) / 2;
        mergeSort(str, l, m);
        mergeSort(str, m + 1, r);
        merge(str, l, m, r);
    }
}

void merge(char str[][10], int l, int m, int r) {
    int i, j, k;
    int n1 = m - l + 1;
    int n2 = r - m;
    char L[n1][10], R[n2][10];

    for (i = 0; i < n1; i++) {
        strcpy(L[i], str[l + i]);
    }

    for (j = 0; j < n2; j++) {
        strcpy(R[j], str[m + 1 + j]);
    }

    i = 0;
    j = 0;
    k = l;

    while (i < n1 && j < n2) {
        if (strcmp(L[i], R[j]) <= 0) {
            strcpy(str[k], L[i]);
            i++;
        } else {
            strcpy(str[k], R[j]);
            j++;
        }
        k++;
    }

    while (i < n1) {
        strcpy(str[k], L[i]);
        i++;
        k++;
    }

    while (j < n2) {
        strcpy(str[k], R[j]);
        j++;
        k++;
    }
}
