#include < stdio.h >
#include < string.h >
#include < time.h >
int selection(char ar[][20], int size, int index) {
	int j;
    if (index == size - 1) {
        return 0;
    }
    for (j = index + 1; j < size; j++) {
        if (strcmp(ar[index], ar[j]) > 0) {
            char temp[100];
            strcpy(temp, ar[index]);
            strcpy(ar[index], ar[j]);
            strcpy(ar[j], temp);
        }
    }

    return selection(ar, size, index + 1);
}

int main() {
	clock_t start=clock();
    char ar[5][20];
    int i;
    clock_t end;
    double cpu_time_used;
    for (i = 0; i < 5; i++) {
    	printf("Enter the strings at %d location\t",i+1);
        scanf("%s", ar[i]);
    }
    selection(ar, 5, 0);
    printf("Sorted Strings are\n");
    for (i = 0; i < 5; i++) {
        printf("%s\n", ar[i]);
    }
    printf("Best and Worst both complexity are same for selection sort\n");
	end=clock();
	cpu_time_used=((double)(end-start))/CLOCKS_PER_SEC;
	printf("Time taken for search: %lf seconds\n",cpu_time_used);
}
