void quick_sort(char a[][10], int low, int high) {
    int pivot;
    if (low < high) {
        pivot = partition(a, low, high);
        quick_sort(a, low, pivot - 1);
        quick_sort(a, pivot + 1, high);
    }
}

int partition(char a[][10], int low, int high) {
    char pivot[10];
    strcpy(pivot, a[high]);
    int i = low - 1;
    for (int j = low; j < high; j++) {
        if (strcmp(a[j], pivot) <= 0) {
            i++;
            step++;
            char temp[10];
            strcpy(temp, a[i]);
            strcpy(a[i], a[j]);
            strcpy(a[j], temp);
        }
    }
    char temp[10];
    strcpy(temp, a[i + 1]);
    strcpy(a[i + 1], a[high]);
    strcpy(a[high], temp);
    return (i + 1);
}
