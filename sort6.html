int step=-1;
int insertion(int ar[], int size, int index) {
	int key=ar[index];
	int j=index-1;
    if (index >= size) {
        return 0;
    }
    while (j >= 0 && ar[j] > key) {
        ar[j + 1] = ar[j];
        j--;
        step++;
       }
    ar[j + 1] = key;
    return insertion(ar, size, index + 1);
}
int main() {

}
