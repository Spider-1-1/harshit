#include < stdio.h >
    #include < string.h >
    #include < time.h >
    void selection(char str[][10],int size)
    {
    int i,j;
    char temp[10];
    for(i=0;i<size;i++)
    {
    for(j=i+1;j<size;j++)
    {
    if(strcmp(str[i],str[j])>0)
    {
    strcpy(temp,str[i]);
    strcpy(str[i],str[j]);
    strcpy(str[j],temp);
    }
    }
    }
    }
    int main()
    {
    clock_t start=clock();
    int i;
    clock_t end;
    double cpu_time_used;
    char str[5][10];
    for(i=0;i<5;i++)
    {
    printf("Enter the strings at %d location\t",i+1);
    scanf("%s",str[i]);
    }
    selection(str,5);
    printf("Sorted strings are\n");
    for(i=0;i<5;i++)
    {
    printf("%s\n",str[i]);
    }
    printf("Best and Worst both complexity are same for selection sort\n");
    end=clock();
    cpu_time_used=((double)(end-start))/CLOCKS_PER_SEC;
    printf("Time taken for search: %lf seconds\n",cpu_time_used);
    }
