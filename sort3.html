    void bubble(int ar[],int size)
    {
    int i,j,temp,step=0;
    for(i=1;i<size;i++)
    {
    for(j=0;j<size-i;j++)
    {
    if(ar[j]>ar[j+1])
    {
    temp=ar[j];
    ar[j]=ar[j+1];
    ar[j+1]=temp;
    }
    }
    }
    }
    int main()
    {
    clock_t start=clock();
    int ar[10],i;
    clock_t end;

    {

    }

    end=clock();
    cpu_time_used=((double)(end-start))/CLOCKS_PER_SEC;
    printf("Time taken for search: %lf seconds\n",cpu_time_used);
    }
