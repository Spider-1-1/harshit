int step=-1;
int insertion(char ar[][10], int size, int index) {
	char temp[10];
	int j;
    if (index >= size) {
        return 0;
    }
    strcpy(temp, ar[index]);
    j = index - 1;
    while (j >= 0 && strcmp(ar[j], temp) > 0) {
        strcpy(ar[j + 1], ar[j]);
        j--;
        step++;
    }
    strcpy(ar[j + 1], temp);
    return insertion(ar, size, index + 1);
}

int main() {
	clock_t start=clock();
    char ar[5][10];

    clock_t end;

    	printf("Enter the string at %d location\t",i+1);
        scanf("%s", ar[i]);
    }
    insertion(ar, 5, 0);

    for (i = 0; i < 5; i++) {
        printf("%s\n", ar[i]);
    }
        if(step==-1)
	{
	printf("Best complexity for selection sort\n");
	}
	else
	{
	printf("Worst complexity for selection sort\n");
	}
	end=clock();
	cpu_time_used=((double)(end-start))/CLOCKS_PER_SEC;
	printf("Time taken for search: %lf seconds\n",cpu_time_used);
}
