#include < stdio.h >
    #include < string.h >
    #include < time.h >
int bubble(char ar[][20], int size, int index) {
	int j;
    if (index == size - 1) {
        return 0;
    }
    for (j = 0; j < size - index - 1; j++) {
        if (strcmp(ar[j], ar[j + 1]) > 0) {
            char temp[100];
            strcpy(temp, ar[j]);
            strcpy(ar[j], ar[j + 1]);
            strcpy(ar[j + 1], temp);
        }
    }
    return bubble(ar, size, index + 1);
}
int main() {
	clock_t start=clock();
    char ar[5][20];
    int i;
    clock_t end;

    	printf("Enter the string at %d location\t",i+1);
        scanf("%s", ar[i]);
    }
    bubble(ar, 5, 0);
    printf("Sorted Strings are\n");
    for (i = 0; i < 5; i++) {
        printf("%s\n", ar[i]);
    }
	printf("Best and Worst both complexity are same for bubble sort\n");
	end=clock();
	cpu_time_used=((double)(end-start))/CLOCKS_PER_SEC;
	printf("Time taken for search: %lf seconds\n",cpu_time_used);
}
