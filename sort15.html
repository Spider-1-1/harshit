#include < stdio.h >
    #include < string.h >
    #include < time.h >
int insertion(char str[][10], int size) {
    int i, j,step=-1;
    char temp[10];

    for (i = 1; i < size; i++) {
        strcpy(temp, str[i]);
        j = i - 1;

        while (j >= 0 && strcmp(temp, str[j]) < 0) {
            strcpy(str[j + 1], str[j]);
            j--;
            step++;
        }

        strcpy(str[j + 1], temp);
    }
    return step;
}

int main() {
	clock_t start=clock();
    char str[5][10];
    int i,step;
    clock_t end;
    double cpu_time_used;
    printf("Enter the strings\n");
    for (i = 0; i < 5; i++) {
        printf("Enter the string at location %d: ", i + 1);
        scanf("%s", str[i]);
    }

    step=insertion(str, 5);

    printf("Sorted Strings are\n");
    for (i = 0; i < 5; i++) {
        printf("%s\n", str[i]);
    }
    if(step==-1)
	{
	printf("Best complexity for selection sort\n");
	}
	else
	{
	printf("Worst complexity for selection sort\n");
	}
	end=clock();
	cpu_time_used=((double)(end-start))/CLOCKS_PER_SEC;
	printf("Time taken for search: %lf seconds\n",cpu_time_used);
}
